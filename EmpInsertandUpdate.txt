package FSS.dao;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;
import java.sql.Statement;
import java.util.Scanner;

public class EmpInsert {

	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		System.out.println("Enter the Driver name:");
		String dr=sc.next();
		try {
		Class.forName(dr);
		System.out.println("load and registering complete");
		System.out.println("Enter the Url:");
		String url=sc.next();
		
			Connection cn=DriverManager.getConnection(url);
			Statement st=cn.createStatement();
			//inserting values into Record
			String query1="insert into project.employeeview values(15,'PVK','pvk@gmail.com',40000,8957393528)";
			//Updating into Record
			String query2="update project.employeeview set empname='Gopalgupta' where empid=11";
			//executing querys
			st.execute(query1);
			st.execute(query2);
			System.out.println("Data inserted");
		} catch (ClassNotFoundException | SQLException e) {
			
			e.printStackTrace();
		}
		
       sc.close();
	}
	}
